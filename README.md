            while i != -1:
                aux[moeda] = i
                if moedas[moeda] * i == 0:
                    makeChange(num, moeda + 1)
                else:
                    makeChange(num - (moedas[moeda] * i), moeda + 1)
                i -= 1
    else:
        aux[moeda] = moedas[moeda] * num
        list.append(aux.copy())
